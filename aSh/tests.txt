// type a
echo hello world

// type a;b;c (multi list)
echo Hello ;   echo There ; echo Who ; echo Are ; echo You ; echo ?

// type a || b || c || d || e && f (multi conditional)
true || echo no || echo no || echo no || echo no && echo yes

// type a && b & c || d (background)
sleep 0.2 && echo Second & sleep 0.1 || echo First

// type a | b | c (multi pipe)
echo Line | cat | wc -l

// type a;b > file (single redirection)
echo Start ; echo File > /temp/redir_0.txt
